    æ‹“æ‰‘æ’åº: ç¡®ä¿å‰ç½®çŸ¥è¯†ç‚¹å…ˆå­¦

        for neighbor in adj_list[current]:

                queue.append(neighbor)

    return [get_knowledge_point(pid) for pid in sorted_list]
